&emsp;&emsp;用选择排序把10个数进行排序，然后输出最大值
<!--more-->

求10个整数中最大值,代码如下
```c 
#include <stdio.h>
#include <stdlib.h>
int main()
{
	int arr1[] = { 2,4,6,5,9,7,1,3,8,10};
	int tmp = 0;
	for (int i = 0; i < sizeof(arr1) / sizeof(arr1[0]); i++)
		for (int j = i + 1; j < sizeof(arr1) / sizeof(arr1[0]); j++)
			if (arr1[i] < arr1[j])
			{
				tmp = arr1[i];
				arr1[i] = arr1[j];
				arr1[j] = tmp;
			}
	printf("%d", arr1[0]);
	system("pause");
	return 0;
}
```
