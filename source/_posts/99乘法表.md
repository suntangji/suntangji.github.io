分别使用双层循环，单层循环，goto语句输出99乘法表
<!--more-->
使用双层循环输出99乘法表
```c
#include<stdio.h>
#include<stdlib.h>
int main()
{
	int i, j;
	for (i = 1; i <= 9; i++)
		for (j = 1; j <= i; j++)
		{
			printf("%d*%d=%d\t", i, j, i*j);
			if (i == j)
				printf("\n");
		}
	system("pause");
	return 0;
}
```
使用一个for循环输出99乘法表
```c
#include <stdio.h>
#include<stdlib.h>
 
int main()
{
	int i = 1,j = 1;
	for (i = 1; i < 10; j++)
	{
		if (j < i)
			printf("%d*%d=%-3d", i, j, i*j);
		else if (j == i)
		{
			printf("%d*%d=%-3d\n", i, j, i*j);
			i++;
			j = 0;
		}
	}
	system("pause");
	return 0;
}
```
用goto 语句实现99乘法表
```
#include <stdio.h>
#include<stdlib.h>
int main()
{
	int i = 0, j = 1;
loop1:
	i++;
	if (i >= 10)
		goto end;
loop2:
	printf("%d*%d=%-3d", i, j, i*j);
	if (j < i)
	{
		j++;
		goto loop2;
	}
	if (j ==i)
	{
		j = 1;
		printf("\n");
		goto loop1;
	}
	
end:
	system("pause");
	return 0;
}
```
